189 __attribute__((always_inline)) inline void __swap_p(void *a, void *b, size_t na, size_t nb)
190 {
191  if ((!na) || (na != nb)) { return; }
192 
193  char *va = (char*)a,
194  *vb = (char*)b;
195 
196  if (va == vb) { return; }
197 
198  __builtin_prefetch(va + na, 1, 1);
199  __builtin_prefetch(vb + nb, 1, 1);
200 
201  while (na--)
202  {
203  va[na] ^= vb[na];
204  vb[na] ^= va[na];

207 }
